# 栈

```
- (void)push:(id)element
{
    if (!items) {
        items = [NSMutableArray new];
    }

    [items addObject:element];
}
- (id)pop
{
    id elenmet = items.lastObject;
    NSLog(@"pop = %@",elenmet);
    [items removeLastObject];
    return elenmet;
}

{
    id elenmet = items.lastObject;
    return elenmet;
}

    }
```

## 优先队列

```

@interface QueueElenmet : NSObject
@property(nonatomic,strong)id element;
@property(nonatomic)NSInteger priority;

@end

- (void)enqueue:(QueueElenmet *)element
{
    if (!items) {
        items = [NSMutableArray new];
    }
    
    if ([self isEmpty]) {
        [items addObject:element];
    }else
    {
        BOOL added = NO;
        for (int i = 0; i < items.count; i++) {
            if (element.priority < ((QueueElenmet *)items[i]).priority) {
                [items insertObject:element atIndex:i];
                added = YES;
                break;
            }
        }
        if (!added) {
            [items addObject:element];
        }
    }
}
```
